    39  func RootFunc(cmd *cobra.Command, args []string) {
    40  	// Parse Config
    41  	var conf *config.CloudgontrollerConfig
    42  	if len(args) == 1 {
    43  		// Parse File and Env
    44  		conf = config.Get(args[0])
    45  	} else {
    46  		// Just parses Environment Variables
    47  		conf = config.Get()
    48  	}
    49  	// Initialize Logger
    50  	logging.Setup(conf)
    51  
    52  	// Initialize Echo Framework
    53  	e := echo.New()
    54  	e.Use(middleware.Recover())
    55  	e.Use(logging.NewEchoZapLogger(zap.L()))
    56  	metrics.EchoPrometheusMiddleware().Use(e)
    57  	e.Use(middleware.RateLimiter(middleware.NewRateLimiterMemoryStoreWithConfig(
    58  		middleware.RateLimiterMemoryStoreConfig{Rate: rate.Limit(conf.RateLimit.Rate), Burst: conf.RateLimit.Burst, ExpiresIn: conf.RateLimit.ExpiresIn},
    59  	)))
    60  
    61  	// Add Validator for Structs
    62  	e.Validator = &api.Validator{Validator: validator.New()}
    63  
    64  	// Initialize DB
    65  	db.NewConnection(conf.DB, true)
    66  	if conf.DB.Migrate {
    67  		db.Migrate(db.GetConnection())
    68  	}
    69  
    70  	// Register API Handlers
    71  	api.RegisterHandlers(e)
    72  
    73  	// Start to Serve
    74  	lock := make(chan error)
    75  	go func(lock chan error) { lock <- e.Start(fmt.Sprintf("%v", conf.Listen)) }(lock)
    76  
    77  	time.Sleep(1 * time.Millisecond)
    78  	zap.L().Warn("Application started without ssl/tls enabled")
    79  
    80  	err := <-lock
    81  	if err != nil {
    82  		zap.L().Panic("failed to start application",zap.Error(err))
    83  	}
    84  }
