    23  		Asset: migrations.ReadFile,
    24  		AssetDir: func() func(string) ([]string, error) {
    25  			return func(path string) ([]string, error) {
    26  				dirEntry, err := migrations.ReadDir(path)
    27  				if err != nil {
    28  					return nil, err
    29  				}
    30  				entries := make([]string, 0)
    31  				for _, e := range dirEntry {
    32  					entries = append(entries, e.Name())
    33  				}
    34  
    35  				return entries, nil
    36  			}
    37  		}(),
    38  		Dir: folder,
    39  	}
    40  
    41  	// Execute the UP Migration
    42  	n, err := migrate.Exec(db, config.Get().DB.Type, migrations, migrate.Up)
    43  	zap.L().Info(fmt.Sprintf("Applied %v DB Migration Steps", n))
    44  	helpers.CheckErrFatal(err)
    45  }
