   715  // Upsert attempts an insert using an executor, and does an update or ignore on conflict.
   716  // See boil.Columns documentation for how to properly use updateColumns and insertColumns.
   717  func (o *Event) Upsert(ctx context.Context, exec boil.ContextExecutor, updateOnConflict bool, conflictColumns []string, updateColumns, insertColumns boil.Columns) error {
   718  	if o == nil {
   719  		return errors.New("models: no events provided for upsert")
   720  	}
   721  	if !boil.TimestampsAreSkipped(ctx) {
   722  		currTime := time.Now().In(boil.GetLocation())
   723  
   724  		if o.CreatedAt.IsZero() {
   725  			o.CreatedAt = currTime
   726  		}
   727  		queries.SetScanner(&o.UpdatedAt, currTime)
   728  	}
   729  
   730  	if err := o.doBeforeUpsertHooks(ctx, exec); err != nil {
   731  		return err
   732  	}
   733  
   734  	nzDefaults := queries.NonZeroDefaultSet(eventColumnsWithDefault, o)
   735  
   736  	// Build cache key in-line uglily - mysql vs psql problems
   737  	buf := strmangle.GetBuffer()
   738  	if updateOnConflict {
   739  		buf.WriteByte('t')
   740  	} else {
   741  		buf.WriteByte('f')
   742  	}
   743  	buf.WriteByte('.')
   744  	for _, c := range conflictColumns {
   745  		buf.WriteString(c)
   746  	}
   747  	buf.WriteByte('.')
   748  	buf.WriteString(strconv.Itoa(updateColumns.Kind))
   749  	for _, c := range updateColumns.Cols {
   750  		buf.WriteString(c)
   751  	}
   752  	buf.WriteByte('.')
   753  	buf.WriteString(strconv.Itoa(insertColumns.Kind))
   754  	for _, c := range insertColumns.Cols {
   755  		buf.WriteString(c)
   756  	}
   757  	buf.WriteByte('.')
   758  	for _, c := range nzDefaults {
   759  		buf.WriteString(c)
   760  	}
   761  	key := buf.String()
   762  	strmangle.PutBuffer(buf)
   763  
   764  	eventUpsertCacheMut.RLock()
   765  	cache, cached := eventUpsertCache[key]
   766  	eventUpsertCacheMut.RUnlock()
   767  
   768  	var err error
   769  
   770  	if !cached {
   771  		insert, ret := insertColumns.InsertColumnSet(
   772  			eventAllColumns,
   773  			eventColumnsWithDefault,
   774  			eventColumnsWithoutDefault,
   775  			nzDefaults,
   776  		)
   777  		update := updateColumns.UpdateColumnSet(
   778  			eventAllColumns,
   779  			eventPrimaryKeyColumns,
   780  		)
   781  
   782  		if updateOnConflict && len(update) == 0 {
   783  			return errors.New("models: unable to upsert events, could not build update column list")
   784  		}
   785  
   786  		conflict := conflictColumns
   787  		if len(conflict) == 0 {
   788  			conflict = make([]string, len(eventPrimaryKeyColumns))
   789  			copy(conflict, eventPrimaryKeyColumns)
   790  		}
   791  		cache.query = buildUpsertQueryPostgres(dialect, "\"events\"", updateOnConflict, ret, update, conflict, insert)
   792  
   793  		cache.valueMapping, err = queries.BindMapping(eventType, eventMapping, insert)
   794  		if err != nil {
   795  			return err
   796  		}
   797  		if len(ret) != 0 {
   798  			cache.retMapping, err = queries.BindMapping(eventType, eventMapping, ret)
   799  			if err != nil {
   800  				return err
   801  			}
   802  		}
   803  	}
   804  
   805  	value := reflect.Indirect(reflect.ValueOf(o))
   806  	vals := queries.ValuesFromMapping(value, cache.valueMapping)
   807  	var returns []interface{}
   808  	if len(cache.retMapping) != 0 {
   809  		returns = queries.PtrsFromMapping(value, cache.retMapping)
   810  	}
   811  
   812  	if boil.IsDebug(ctx) {
   813  		writer := boil.DebugWriterFrom(ctx)
   814  		fmt.Fprintln(writer, cache.query)
   815  		fmt.Fprintln(writer, vals)
   816  	}
   817  	if len(cache.retMapping) != 0 {
   818  		err = exec.QueryRowContext(ctx, cache.query, vals...).Scan(returns...)
   819  		if err == sql.ErrNoRows {
   820  			err = nil // Postgres doesn't return anything when there's no update
   821  		}
   822  	} else {
   823  		_, err = exec.ExecContext(ctx, cache.query, vals...)
   824  	}
   825  	if err != nil {
   826  		return errors.Wrap(err, "models: unable to upsert events")
   827  	}
   828  
   829  	if !cached {
   830  		eventUpsertCacheMut.Lock()
   831  		eventUpsertCache[key] = cache
   832  		eventUpsertCacheMut.Unlock()
   833  	}
   834  
   835  	return o.doAfterUpsertHooks(ctx, exec)
   836  }
