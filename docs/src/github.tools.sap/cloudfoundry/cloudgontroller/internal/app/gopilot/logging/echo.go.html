    12  // ZapLogger is a middleware and zap to provide an "access log" like logging for each request.
    13  func NewEchoZapLogger(log *zap.Logger) echo.MiddlewareFunc {
    14  	return func(next echo.HandlerFunc) echo.HandlerFunc {
    15  		return func(c echo.Context) error {
    16  			start := time.Now()
    17  
    18  			req := c.Request()
    19  			fields := []zapcore.Field{
    20  				zap.String("remote_ip", c.RealIP()),
    21  				zap.String("time", time.Since(start).String()),
    22  				zap.String("host", req.Host),
    23  				zap.String("request", fmt.Sprintf("%s %s", req.Method, req.RequestURI)),
    24  				zap.String("user_agent", req.UserAgent()),
    25  			}
    26  
    27  			log.Debug("Request recieved", fields...)
    28  
    29  			err := next(c)
    30  			if err != nil {
    31  				log = log.With(zap.Error(err))
    32  				c.Error(err)
    33  			}
    34  
    35  			res := c.Response()
    36  
    37  			fields = []zapcore.Field{
    38  				zap.String("remote_ip", c.RealIP()),
    39  				zap.String("time", time.Since(start).String()),
    40  				zap.String("host", req.Host),
    41  				zap.String("request", fmt.Sprintf("%s %s", req.Method, req.RequestURI)),
    42  				zap.Int("status", res.Status),
    43  				zap.Int64("size", res.Size),
    44  				zap.String("user_agent", req.UserAgent()),
    45  			}
    46  
    47  			id := req.Header.Get(echo.HeaderXRequestID)
    48  			if id == "" {
    49  				id = res.Header().Get(echo.HeaderXRequestID)
    50  				fields = append(fields, zap.String("request_id", id))
    51  			}
    52  
    53  			n := res.Status
    54  			switch {
    55  			case n >= 500:
    56  				log.Error("Server error", fields...)
    57  			case n >= 400:
    58  				log.Warn("Client error", fields...)
    59  			case n >= 300:
    60  				log.Info("Redirection", fields...)
    61  			default:
    62  				log.Info("Success", fields...)
    63  			}
    64  
    65  			return nil
    66  		}
    67  	}
    68  }
