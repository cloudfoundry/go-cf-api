   602  // Upsert attempts an insert using an executor, and does an update or ignore on conflict.
   603  // See boil.Columns documentation for how to properly use updateColumns and insertColumns.
   604  func (o *GorpMigration) Upsert(ctx context.Context, exec boil.ContextExecutor, updateOnConflict bool, conflictColumns []string, updateColumns, insertColumns boil.Columns) error {
   605  	if o == nil {
   606  		return errors.New("models: no gorp_migrations provided for upsert")
   607  	}
   608  
   609  	if err := o.doBeforeUpsertHooks(ctx, exec); err != nil {
   610  		return err
   611  	}
   612  
   613  	nzDefaults := queries.NonZeroDefaultSet(gorpMigrationColumnsWithDefault, o)
   614  
   615  	// Build cache key in-line uglily - mysql vs psql problems
   616  	buf := strmangle.GetBuffer()
   617  	if updateOnConflict {
   618  		buf.WriteByte('t')
   619  	} else {
   620  		buf.WriteByte('f')
   621  	}
   622  	buf.WriteByte('.')
   623  	for _, c := range conflictColumns {
   624  		buf.WriteString(c)
   625  	}
   626  	buf.WriteByte('.')
   627  	buf.WriteString(strconv.Itoa(updateColumns.Kind))
   628  	for _, c := range updateColumns.Cols {
   629  		buf.WriteString(c)
   630  	}
   631  	buf.WriteByte('.')
   632  	buf.WriteString(strconv.Itoa(insertColumns.Kind))
   633  	for _, c := range insertColumns.Cols {
   634  		buf.WriteString(c)
   635  	}
   636  	buf.WriteByte('.')
   637  	for _, c := range nzDefaults {
   638  		buf.WriteString(c)
   639  	}
   640  	key := buf.String()
   641  	strmangle.PutBuffer(buf)
   642  
   643  	gorpMigrationUpsertCacheMut.RLock()
   644  	cache, cached := gorpMigrationUpsertCache[key]
   645  	gorpMigrationUpsertCacheMut.RUnlock()
   646  
   647  	var err error
   648  
   649  	if !cached {
   650  		insert, ret := insertColumns.InsertColumnSet(
   651  			gorpMigrationAllColumns,
   652  			gorpMigrationColumnsWithDefault,
   653  			gorpMigrationColumnsWithoutDefault,
   654  			nzDefaults,
   655  		)
   656  		update := updateColumns.UpdateColumnSet(
   657  			gorpMigrationAllColumns,
   658  			gorpMigrationPrimaryKeyColumns,
   659  		)
   660  
   661  		if updateOnConflict && len(update) == 0 {
   662  			return errors.New("models: unable to upsert gorp_migrations, could not build update column list")
   663  		}
   664  
   665  		conflict := conflictColumns
   666  		if len(conflict) == 0 {
   667  			conflict = make([]string, len(gorpMigrationPrimaryKeyColumns))
   668  			copy(conflict, gorpMigrationPrimaryKeyColumns)
   669  		}
   670  		cache.query = buildUpsertQueryPostgres(dialect, "\"gorp_migrations\"", updateOnConflict, ret, update, conflict, insert)
   671  
   672  		cache.valueMapping, err = queries.BindMapping(gorpMigrationType, gorpMigrationMapping, insert)
   673  		if err != nil {
   674  			return err
   675  		}
   676  		if len(ret) != 0 {
   677  			cache.retMapping, err = queries.BindMapping(gorpMigrationType, gorpMigrationMapping, ret)
   678  			if err != nil {
   679  				return err
   680  			}
   681  		}
   682  	}
   683  
   684  	value := reflect.Indirect(reflect.ValueOf(o))
   685  	vals := queries.ValuesFromMapping(value, cache.valueMapping)
   686  	var returns []interface{}
   687  	if len(cache.retMapping) != 0 {
   688  		returns = queries.PtrsFromMapping(value, cache.retMapping)
   689  	}
   690  
   691  	if boil.IsDebug(ctx) {
   692  		writer := boil.DebugWriterFrom(ctx)
   693  		fmt.Fprintln(writer, cache.query)
   694  		fmt.Fprintln(writer, vals)
   695  	}
   696  	if len(cache.retMapping) != 0 {
   697  		err = exec.QueryRowContext(ctx, cache.query, vals...).Scan(returns...)
   698  		if err == sql.ErrNoRows {
   699  			err = nil // Postgres doesn't return anything when there's no update
   700  		}
   701  	} else {
   702  		_, err = exec.ExecContext(ctx, cache.query, vals...)
   703  	}
   704  	if err != nil {
   705  		return errors.Wrap(err, "models: unable to upsert gorp_migrations")
   706  	}
   707  
   708  	if !cached {
   709  		gorpMigrationUpsertCacheMut.Lock()
   710  		gorpMigrationUpsertCache[key] = cache
   711  		gorpMigrationUpsertCacheMut.Unlock()
   712  	}
   713  
   714  	return o.doAfterUpsertHooks(ctx, exec)
   715  }
