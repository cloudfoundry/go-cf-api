    55  // Parses the config once, otherwise just returns it
    56  // Priority from lowest to highest: Default Values, Config File, Environment Variables
    57  // After setting the Config in this order it is validated according to struct tags
    58  // If validation is not succefull we have a invalid config and thus throw a FATAL
    59  func Get(params ...string) *CloudgontrollerConfig {
    60  	once.Do(func() {
    61  		tmp := &CloudgontrollerConfig{}
    62  		// Default Values
    63  		helpers.CheckErrFatal(defaults.Set(tmp))
    64  		// If Config File is present, Parse it into a struct
    65  		if len(params) == 1 {
    66  			viper.SetConfigFile(params[0])
    67  			err := viper.ReadInConfig()
    68  			if err == nil {
    69  				zap.L().Info(fmt.Sprintf("Using config file %s", viper.ConfigFileUsed()), zap.String(log.Tags.File, viper.ConfigFileUsed()))
    70  			} else {
    71  				helpers.CheckErrFatal(err)
    72  			}
    73  			err = viper.Unmarshal(tmp)
    74  			helpers.CheckErrFatal(err)
    75  		}
    76  		// Overwrite values with env Variables
    77  		helpers.CheckErrFatal(env.Parse(tmp))
    78  		if zap.L().Core().Enabled(zap.DebugLevel) {
    79  			usedConfig, err := yaml.Marshal(config)
    80  			helpers.CheckErrFatal(err)
    81  			zap.L().Debug("Using following config", zap.String("Config", string(usedConfig)))
    82  		}
    83  		// Validate the Result if it is valid
    84  		helpers.CheckErrFatal(tmp.Validate())
    85  		config = tmp
    86  	})
    87  	return config
    88  }

    95  // We need to extend the validation function because of https://github.com/go-playground/validator/issues/782
    96  // Otherwise users would not get a clue why their config did not validate (meaningless error message)
    97  func (s *CloudgontrollerConfig) Validate() error {
    98  	// Check config with validator tags
    99  	if err := validator.New().Struct(s); err != nil {
   100  		fields := strings.Split(strings.Split(err.Error(), "'")[1], ".")
   101  		currentField := reflect.TypeOf(s).Elem()
   102  		var resultField reflect.StructField
   103  		// Runtime is O(1) for this loop, as we use FieldByName
   104  		for fieldNumber, fieldName := range fields {
   105  			// The first field is the struct name, we must skip that
   106  			if fieldNumber > 0 {
   107  				f, _ := currentField.FieldByName(fieldName)
   108  				if f.Type.Kind() == reflect.Struct {
   109  					// If the next field is a struct, we itterate deeper
   110  					currentField = f.Type
   111  				} else {
   112  					// If it is not we have a basic datatype so we save that field
   113  					resultField = f
   114  					break
   115  				}
   116  			}
   117  		}
   118  		return fmt.Errorf("error parsing config: %s. values must comply with following schema: \"%s\". Read more what the validate tags mean here: https://github.com/go-playground/validator ", err, resultField.Tag.Get("validate"))
   119  	}
   120  
   121  	// Check the DB Connection String
   122  	if s.DB.Type == "postgres" {
   123  		_, err := pgx.ParseConfig(s.DB.ConnectionString)
   124  		if err != nil {
   125  			return fmt.Errorf("invalid config.db.connectionstring: %v", err)
   126  		}
   127  	} else {
   128  		_, err := mysql.ParseDSN(s.DB.ConnectionString)
   129  		if err != nil {
   130  			return fmt.Errorf("invalid config.db.connectionstring: %v", err)
   131  		}
   132  	}
   133  	return nil
   134  }
