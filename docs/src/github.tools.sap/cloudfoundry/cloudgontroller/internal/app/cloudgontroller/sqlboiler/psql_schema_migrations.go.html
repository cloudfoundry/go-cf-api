   594  // Upsert attempts an insert using an executor, and does an update or ignore on conflict.
   595  // See boil.Columns documentation for how to properly use updateColumns and insertColumns.
   596  func (o *SchemaMigration) Upsert(ctx context.Context, exec boil.ContextExecutor, updateOnConflict bool, conflictColumns []string, updateColumns, insertColumns boil.Columns) error {
   597  	if o == nil {
   598  		return errors.New("models: no schema_migrations provided for upsert")
   599  	}
   600  
   601  	if err := o.doBeforeUpsertHooks(ctx, exec); err != nil {
   602  		return err
   603  	}
   604  
   605  	nzDefaults := queries.NonZeroDefaultSet(schemaMigrationColumnsWithDefault, o)
   606  
   607  	// Build cache key in-line uglily - mysql vs psql problems
   608  	buf := strmangle.GetBuffer()
   609  	if updateOnConflict {
   610  		buf.WriteByte('t')
   611  	} else {
   612  		buf.WriteByte('f')
   613  	}
   614  	buf.WriteByte('.')
   615  	for _, c := range conflictColumns {
   616  		buf.WriteString(c)
   617  	}
   618  	buf.WriteByte('.')
   619  	buf.WriteString(strconv.Itoa(updateColumns.Kind))
   620  	for _, c := range updateColumns.Cols {
   621  		buf.WriteString(c)
   622  	}
   623  	buf.WriteByte('.')
   624  	buf.WriteString(strconv.Itoa(insertColumns.Kind))
   625  	for _, c := range insertColumns.Cols {
   626  		buf.WriteString(c)
   627  	}
   628  	buf.WriteByte('.')
   629  	for _, c := range nzDefaults {
   630  		buf.WriteString(c)
   631  	}
   632  	key := buf.String()
   633  	strmangle.PutBuffer(buf)
   634  
   635  	schemaMigrationUpsertCacheMut.RLock()
   636  	cache, cached := schemaMigrationUpsertCache[key]
   637  	schemaMigrationUpsertCacheMut.RUnlock()
   638  
   639  	var err error
   640  
   641  	if !cached {
   642  		insert, ret := insertColumns.InsertColumnSet(
   643  			schemaMigrationAllColumns,
   644  			schemaMigrationColumnsWithDefault,
   645  			schemaMigrationColumnsWithoutDefault,
   646  			nzDefaults,
   647  		)
   648  		update := updateColumns.UpdateColumnSet(
   649  			schemaMigrationAllColumns,
   650  			schemaMigrationPrimaryKeyColumns,
   651  		)
   652  
   653  		if updateOnConflict && len(update) == 0 {
   654  			return errors.New("models: unable to upsert schema_migrations, could not build update column list")
   655  		}
   656  
   657  		conflict := conflictColumns
   658  		if len(conflict) == 0 {
   659  			conflict = make([]string, len(schemaMigrationPrimaryKeyColumns))
   660  			copy(conflict, schemaMigrationPrimaryKeyColumns)
   661  		}
   662  		cache.query = buildUpsertQueryPostgres(dialect, "\"schema_migrations\"", updateOnConflict, ret, update, conflict, insert)
   663  
   664  		cache.valueMapping, err = queries.BindMapping(schemaMigrationType, schemaMigrationMapping, insert)
   665  		if err != nil {
   666  			return err
   667  		}
   668  		if len(ret) != 0 {
   669  			cache.retMapping, err = queries.BindMapping(schemaMigrationType, schemaMigrationMapping, ret)
   670  			if err != nil {
   671  				return err
   672  			}
   673  		}
   674  	}
   675  
   676  	value := reflect.Indirect(reflect.ValueOf(o))
   677  	vals := queries.ValuesFromMapping(value, cache.valueMapping)
   678  	var returns []interface{}
   679  	if len(cache.retMapping) != 0 {
   680  		returns = queries.PtrsFromMapping(value, cache.retMapping)
   681  	}
   682  
   683  	if boil.IsDebug(ctx) {
   684  		writer := boil.DebugWriterFrom(ctx)
   685  		fmt.Fprintln(writer, cache.query)
   686  		fmt.Fprintln(writer, vals)
   687  	}
   688  	if len(cache.retMapping) != 0 {
   689  		err = exec.QueryRowContext(ctx, cache.query, vals...).Scan(returns...)
   690  		if err == sql.ErrNoRows {
   691  			err = nil // Postgres doesn't return anything when there's no update
   692  		}
   693  	} else {
   694  		_, err = exec.ExecContext(ctx, cache.query, vals...)
   695  	}
   696  	if err != nil {
   697  		return errors.Wrap(err, "models: unable to upsert schema_migrations")
   698  	}
   699  
   700  	if !cached {
   701  		schemaMigrationUpsertCacheMut.Lock()
   702  		schemaMigrationUpsertCache[key] = cache
   703  		schemaMigrationUpsertCacheMut.Unlock()
   704  	}
   705  
   706  	return o.doAfterUpsertHooks(ctx, exec)
   707  }
