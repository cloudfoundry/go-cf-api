    17  func TestZapLogger(t *testing.T) {
    18  	e := echo.New()
    19  	req := httptest.NewRequest(http.MethodGet, "/something", nil)
    20  	rec := httptest.NewRecorder()
    21  	c := e.NewContext(req, rec)
    22  
    23  	h := func(c echo.Context) error {
    24  		return c.String(http.StatusOK, "")
    25  	}
    26  
    27  	obs, logs := observer.New(zap.DebugLevel)
    28  
    29  	logger := zap.New(obs)
    30  
    31  	err := logging.NewEchoZapLogger(logger)(h)(c)
    32  
    33  	assert.Nil(t, err)
    34  	assert.Equal(t, 2, logs.Len())
    35  
    36  	logFieldsStart := logs.AllUntimed()[0].ContextMap()
    37  	assert.Equal(t, "example.com", logFieldsStart["host"])
    38  	assert.NotNil(t, logFieldsStart["time"])
    39  	assert.Equal(t, "GET /something", logFieldsStart["request"])
    40  	assert.NotNil(t, logFieldsStart["user_agent"])
    41  
    42  	logFieldsEnd := logs.AllUntimed()[1].ContextMap()
    43  	assert.Equal(t, "192.0.2.1", logFieldsEnd["remote_ip"])
    44  	assert.Equal(t, "example.com", logFieldsEnd["host"])
    45  	assert.NotNil(t, logFieldsEnd["time"])
    46  	assert.Equal(t, "GET /something", logFieldsEnd["request"])
    47  	assert.Equal(t, int64(200), logFieldsEnd["status"])
    48  	assert.Equal(t, int64(0), logFieldsEnd["size"])
    49  	assert.NotNil(t, logFieldsEnd["user_agent"])
    50  	assert.NotNil(t, logFieldsEnd["request_id"])
    51  }
