    15  func TestZapLogger(t *testing.T) {
    16  	e := echo.New()
    17  	req := httptest.NewRequest(http.MethodGet, "/something", nil)
    18  	rec := httptest.NewRecorder()
    19  	c := e.NewContext(req, rec)
    20  
    21  	h := func(c echo.Context) error {
    22  		return c.String(http.StatusOK, "")
    23  	}
    24  
    25  	obs, logs := observer.New(zap.DebugLevel)
    26  
    27  	logger := zap.New(obs)
    28  
    29  	err := logging.NewEchoZapLogger(logger)(h)(c)
    30  
    31  	assert.Nil(t, err)
    32  	assert.Equal(t, 2, logs.Len())
    33  
    34  	logFieldsStart := logs.AllUntimed()[0].ContextMap()
    35  	assert.Equal(t, "example.com", logFieldsStart["host"])
    36  	assert.NotNil(t, logFieldsStart["time"])
    37  	assert.Equal(t, "GET /something", logFieldsStart["request"])
    38  	assert.NotNil(t, logFieldsStart["user_agent"])
    39  
    40  	logFieldsEnd := logs.AllUntimed()[1].ContextMap()
    41  	assert.Equal(t, "192.0.2.1", logFieldsEnd["remote_ip"])
    42  	assert.Equal(t, "example.com", logFieldsEnd["host"])
    43  	assert.NotNil(t, logFieldsEnd["time"])
    44  	assert.Equal(t, "GET /something", logFieldsEnd["request"])
    45  	assert.Equal(t, int64(200), logFieldsEnd["status"])
    46  	assert.Equal(t, int64(0), logFieldsEnd["size"])
    47  	assert.NotNil(t, logFieldsEnd["user_agent"])
